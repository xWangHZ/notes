# 快速幂

```c++
#include<bits/stdc++.h>
using namespace std;
int pow(int a,int b)
{
    int ans = 1,base = a;// ans:幂的结果；base:底数a
    while(b)
    {
        if(b&1)
        {
            ans = ans*base;
        }
        base = base * base;
        b = b >> 1;
    }
    return ans;
}
int main(){
    int a,b;
    cin>>a>>b;
    cout<<pow(a,b)<<endl;
    return 0;
}
```
